 public class TimeUtils {
 
     public static int timeToSec(String time) {
         String[] part = time.split(":");
         int hh = Integer.parseInt(part[0]);
         int mm = Integer.parseInt(part[1]);
         int ss = Integer.parseInt(part[2]);
 
         return 3600 * hh + 60 * mm + ss;
     }
